¿Cómo conectar tu App?

Pega esto al principio de tu archivo www/js/api_db.js para automatizar la conexión. 

// 1. Leer configuración del CPANEL (o usar la de fábrica)
const savedEnv = localStorage.getItem('APP_ENV') || 'dev';
const devUrl = localStorage.getItem('APP_DEV_URL') || 'https://desktop-71uhcug.tailb81e5f.ts.net';
const prodUrl = localStorage.getItem('APP_PROD_URL') || 'https://api.tu-app.com';

// 2. Asignar URL dinámica
const POCKETBASE_URL = savedEnv === 'prod' ? prodUrl : devUrl;
const APP_SLUG = savedEnv === 'prod' ? 'App_Oficial' : 'App_Pruebas';

console.log("🔥 Entorno:", savedEnv.toUpperCase());
console.log("📡 Conectando:", POCKETBASE_URL);
